application.yml

resource目录下新建application.yml配置文件，配置一下信息：

server:
  port: 8095
spring:
  application:
    name: service-producer
eureka:
  client:
    service-url:
      defaultZone: http://localhost:8091/eureka,http://localhost:8092/eureka,http://localhost:8093/eureka,

EurekaProducerApplication.java

package com.zms.producer;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.cloud.client.discovery.EnableDiscoveryClient;
/**
 * @author zongman
 */
@SpringBootApplication
@EnableDiscoveryClient
public class EurekaProducerApplication {
    public static void main(String[] args) {
        SpringApplication.run(EurekaProducerApplication.class, args);
    }
}

添加@EnableDiscoveryClient注解后，项目就具有了服务注册的功能。启动工程后，就可以在注册中心的页面看到SERVICE-PRODUCER服务。
